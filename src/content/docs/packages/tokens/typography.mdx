# Typography Tokens

Typography tokens define the text styles used throughout the design system, ensuring consistent and accessible text rendering.

## Font Families

### JavaScript (Web Components)

```javascript
// Apply typography tokens to elements
const heading = document.createElement('h1');
heading.style.setProperty('--font-family', 'var(--token-font-family-heading)');
heading.style.setProperty('--font-size', 'var(--token-font-size-2xl)');
heading.style.setProperty('--font-weight', 'var(--token-font-weight-bold)');
heading.style.setProperty('--line-height', 'var(--token-line-height-tight)');

// Create a text component
class TypographyElement extends HTMLElement {
  connectedCallback() {
    const variant = this.getAttribute('variant') || 'body';
    this.style.cssText = `
      font-family: var(--token-font-family-${variant});
      font-size: var(--token-font-size-${variant});
      line-height: var(--token-line-height-${variant});
    `;
  }
}

customElements.define('typography-element', TypographyElement);
```

### React

```tsx
import { tokens } from '@your-package/tokens';

interface TypographyProps {
  variant: 'h1' | 'h2' | 'h3' | 'body' | 'caption';
  children: React.ReactNode;
  className?: string;
}

function Typography({ variant, children, className }: TypographyProps) {
  const getTypographyStyles = (variant: string) => {
    const baseStyles = {
      fontFamily: tokens.typography.fontFamily.base,
      margin: 0,
    };

    switch (variant) {
      case 'h1':
        return {
          ...baseStyles,
          fontSize: tokens.typography.fontSize['4xl'],
          fontWeight: tokens.typography.fontWeight.bold,
          lineHeight: tokens.typography.lineHeight.tight,
        };
      case 'h2':
        return {
          ...baseStyles,
          fontSize: tokens.typography.fontSize['3xl'],
          fontWeight: tokens.typography.fontWeight.semibold,
          lineHeight: tokens.typography.lineHeight.snug,
        };
      case 'body':
        return {
          ...baseStyles,
          fontSize: tokens.typography.fontSize.base,
          fontWeight: tokens.typography.fontWeight.normal,
          lineHeight: tokens.typography.lineHeight.relaxed,
        };
      default:
        return baseStyles;
    }
  };

  const Component = variant.startsWith('h') ? variant : 'p';
  
  return (
    <Component 
      style={getTypographyStyles(variant)}
      className={className}
    >
      {children}
    </Component>
  );
}
```

## Font Sizes

Our typography scale provides consistent sizing across all text elements.

```css
/* Font Size Tokens */
:root {
  --token-font-size-xs: 0.75rem;    /* 12px */
  --token-font-size-sm: 0.875rem;   /* 14px */
  --token-font-size-base: 1rem;     /* 16px */
  --token-font-size-lg: 1.125rem;   /* 18px */
  --token-font-size-xl: 1.25rem;    /* 20px */
  --token-font-size-2xl: 1.5rem;    /* 24px */
  --token-font-size-3xl: 1.875rem;  /* 30px */
  --token-font-size-4xl: 2.25rem;   /* 36px */
}
```

## Line Heights

Consistent line heights improve readability and visual hierarchy.

| Token | Value | Usage |
|-------|-------|-------|
| `--token-line-height-tight` | `1.25` | Headings |
| `--token-line-height-snug` | `1.375` | Subheadings |
| `--token-line-height-normal` | `1.5` | Body text |
| `--token-line-height-relaxed` | `1.625` | Long-form content |
