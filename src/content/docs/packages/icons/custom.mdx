# Custom Icons

Learn how to create and integrate custom icons into your design system.

## Creating Custom Icons

### Design Guidelines

When creating custom icons, follow these design principles:

- **Consistency** - Match the visual style of existing icons
- **Simplicity** - Keep designs clean and recognizable at small sizes
- **Clarity** - Ensure icons are easily distinguishable from each other

### JavaScript (Web Components)

```javascript
// Custom icon registry
class CustomIconRegistry {
  constructor() {
    this.icons = new Map();
  }

  register(name, pathData, viewBox = '0 0 24 24') {
    this.icons.set(name, { pathData, viewBox });
  }

  getIcon(name) {
    return this.icons.get(name);
  }

  createIcon(name, size = 24, color = 'currentColor') {
    const icon = this.getIcon(name);
    if (!icon) {
      console.warn(`Icon "${name}" not found`);
      return null;
    }

    const svg = document.createElementNS('http://www.w3.org/2000/svg', 'svg');
    svg.setAttribute('width', size);
    svg.setAttribute('height', size);
    svg.setAttribute('viewBox', icon.viewBox);
    svg.setAttribute('fill', color);
    svg.innerHTML = `<path d="${icon.pathData}"/>`;
    
    return svg;
  }
}

// Usage
const iconRegistry = new CustomIconRegistry();

// Register custom icons
iconRegistry.register('bitcoin', 'M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zm-1 17.93c-3.94-.49-7-3.85-7-7.93 0-.62.08-1.21.21-1.79L9 15v1c0 1.1.9 2 2 2v1.93zm6.9-2.54c-.26-.81-1-1.39-1.9-1.39h-1v-3c0-.55-.45-1-1-1H8v-2h2c.55 0 1-.45 1-1V7h2c1.1 0 2-.9 2-2v-.41c2.93 1.19 5 4.06 5 7.41 0 2.08-.8 3.97-2.1 5.39z');
iconRegistry.register('lightning', 'M7 2v11h3v9l7-12h-4l4-8z');

// Create custom icon component
class CustomIcon extends HTMLElement {
  static get observedAttributes() {
    return ['name', 'size', 'color'];
  }

  connectedCallback() {
    this.render();
  }

  attributeChangedCallback() {
    this.render();
  }

  render() {
    const name = this.getAttribute('name');
    const size = this.getAttribute('size') || '24';
    const color = this.getAttribute('color') || 'currentColor';
    
    const icon = iconRegistry.createIcon(name, size, color);
    if (icon) {
      this.innerHTML = '';
      this.appendChild(icon);
    }
  }
}

customElements.define('custom-icon', CustomIcon);
```

### React

```tsx
import { createContext, useContext, ReactNode } from 'react';

// Custom icon context
interface CustomIconData {
  pathData: string;
  viewBox: string;
}

interface IconRegistryContextType {
  registerIcon: (name: string, data: CustomIconData) => void;
  getIcon: (name: string) => CustomIconData | undefined;
}

const IconRegistryContext = createContext<IconRegistryContextType | null>(null);

// Icon registry provider
function IconRegistryProvider({ children }: { children: ReactNode }) {
  const icons = new Map<string, CustomIconData>();

  const registerIcon = (name: string, data: CustomIconData) => {
    icons.set(name, data);
  };

  const getIcon = (name: string) => {
    return icons.get(name);
  };

  return (
    <IconRegistryContext.Provider value={{ registerIcon, getIcon }}>
      {children}
    </IconRegistryContext.Provider>
  );
}

// Custom icon component
interface CustomIconProps {
  name: string;
  size?: number | string;
  color?: string;
  className?: string;
}

function CustomIcon({ name, size = 24, color = 'currentColor', className }: CustomIconProps) {
  const { getIcon } = useContext(IconRegistryContext)!;
  const icon = getIcon(name);

  if (!icon) {
    console.warn(`Custom icon "${name}" not found`);
    return null;
  }

  return (
    <svg
      width={size}
      height={size}
      viewBox={icon.viewBox}
      fill={color}
      className={className}
    >
      <path d={icon.pathData} />
    </svg>
  );
}

// Usage
function App() {
  return (
    <IconRegistryProvider>
      <div>
        <CustomIcon name="bitcoin" size={32} color="#f7931a" />
        <CustomIcon name="lightning" size={24} color="#ffd700" />
      </div>
    </IconRegistryProvider>
  );
}
```

## SVG Optimization

Optimize your SVG icons for better performance and consistency.

```javascript
// SVG optimization utility
function optimizeSVG(svgString) {
  // Remove unnecessary attributes
  const cleaned = svgString
    .replace(/fill="[^"]*"/g, '') // Remove fill attributes (use CSS)
    .replace(/stroke="[^"]*"/g, '') // Remove stroke attributes
    .replace(/\s+/g, ' ') // Minimize whitespace
    .trim();

  return cleaned;
}

// Convert SVG to icon data
function svgToIconData(svgElement) {
  const viewBox = svgElement.getAttribute('viewBox') || '0 0 24 24';
  const path = svgElement.querySelector('path');
  const pathData = path ? path.getAttribute('d') : '';
  
  return { pathData, viewBox };
}
```

## Integration

Add custom icons to your existing icon system.

```tsx
// Extend the main icon component
import { Icon } from '@your-package/icons';

interface ExtendedIconProps {
  name: string;
  size?: number | string;
  color?: string;
  custom?: boolean;
}

function ExtendedIcon({ name, size = 24, color = 'currentColor', custom = false }: ExtendedIconProps) {
  if (custom) {
    return <CustomIcon name={name} size={size} color={color} />;
  }
  
  return <Icon name={name} size={size} color={color} />;
}
```
