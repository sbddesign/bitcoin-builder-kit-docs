# Icons

A comprehensive icon system providing consistent, accessible icons for all your applications.

## What's Included

- **Icon Library** - 500+ carefully crafted icons
- **Usage Guidelines** - Best practices for icon implementation
- **Custom Icons** - Tools for creating and adding custom icons

## Getting Started

Icons are available as both SVG components and icon fonts, with support for multiple frameworks.

## Usage Examples

### JavaScript (Web Components)

```javascript
// Create an icon component
class IconComponent extends HTMLElement {
  static get observedAttributes() {
    return ['name', 'size', 'color'];
  }

  connectedCallback() {
    this.render();
  }

  attributeChangedCallback() {
    this.render();
  }

  render() {
    const name = this.getAttribute('name') || 'star';
    const size = this.getAttribute('size') || '24';
    const color = this.getAttribute('color') || 'currentColor';
    
    this.innerHTML = `
      <svg width="${size}" height="${size}" viewBox="0 0 24 24" fill="${color}">
        ${this.getIconPath(name)}
      </svg>
    `;
  }

  getIconPath(name) {
    const icons = {
      star: '<path d="M12 2l3.09 6.26L22 9.27l-5 4.87 1.18 6.88L12 17.77l-6.18 3.25L7 14.14 2 9.27l6.91-1.01L12 2z"/>',
      heart: '<path d="M20.84 4.61a5.5 5.5 0 0 0-7.78 0L12 5.67l-1.06-1.06a5.5 5.5 0 0 0-7.78 7.78l1.06 1.06L12 21.23l7.78-7.78 1.06-1.06a5.5 5.5 0 0 0 0-7.78z"/>',
      // Add more icons...
    };
    return icons[name] || icons.star;
  }
}

customElements.define('icon-component', IconComponent);

// Usage
const icon = document.createElement('icon-component');
icon.setAttribute('name', 'heart');
icon.setAttribute('size', '32');
icon.setAttribute('color', '#ef4444');
```

### React

```tsx
import { Icon } from '@your-package/icons';

interface IconProps {
  name: string;
  size?: number | string;
  color?: string;
  className?: string;
}

function Icon({ name, size = 24, color = 'currentColor', className }: IconProps) {
  return (
    <svg 
      width={size} 
      height={size} 
      viewBox="0 0 24 24" 
      fill={color}
      className={className}
      aria-hidden="true"
    >
      <path d={getIconPath(name)} />
    </svg>
  );
}

// Usage examples
function App() {
  return (
    <div>
      <Icon name="star" size={24} color="#fbbf24" />
      <Icon name="heart" size={32} color="#ef4444" />
      <Icon name="user" size="1.5rem" />
    </div>
  );
}
```
